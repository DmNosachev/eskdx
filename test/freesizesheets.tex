\newcommand{\printalot}[2]{
  \setcounter{printalotfirst}{0}
  \setcounter{printalotsecond}{#1}
  \whiledo{\value{printalotfirst}<\value{printalotsecond}}
  {#2 \stepcounter{printalotfirst}}
}
